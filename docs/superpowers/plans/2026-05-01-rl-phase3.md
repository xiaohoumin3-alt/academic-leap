```typescript
  config: T;

function parseEnvBool(key: string, defaultValue: boolean): boolean {
  const value = process.env[key];
  if (value === undefined) return defaultValue;
  return value === 'true' || value === '1';
}

function parseEnvNumber(key: string, defaultValue: number): number {
  const value = process.env[key];
  if (value === undefined) return defaultValue;
  const parsed = Number(value);
  return isNaN(parsed) ? defaultValue : parsed;
}
```

```typescript
// lib/rl/quality/label-quality.ts

import type { LQMConfig } from '../config/phase3-features';
import type { QuestionQuality, CorrectedLabel, QuestionAttempt, StudentResponse } from './types';

/**
 * Label Quality Model - Estimates question label quality and corrects noisy labels
 *
 * Uses simplified IRT model extension:
 * P(correct) = quality * IRT_prob + (1-quality) * 0.5
 *
 * Where quality is the probability that the label is correct
 */
export class LabelQualityModel {
  private readonly config: LQMConfig;
  private questionAttempts: Map<string, QuestionAttempt[]> = new Map();

  constructor(config: LQMConfig) {
    this.config = config;
  }

  /**
   * Estimate question quality based on response history
   */
  estimateQuality(questionId: string, history: QuestionAttempt[], difficulty: number): QuestionQuality {
    const attempts = this.questionAttempts.get(questionId) ?? history;

    if (attempts.length === 0) {
      return {
        questionId,
        estimatedQuality: 1.0,
        confidence: 0,
        isNoisy: false,
      };
    }

    // Calculate average theta
    const avgTheta = attempts.reduce((sum, a) => sum + a.theta, 0) / attempts.length;

    // Calculate correct rate
    const correctRate = attempts.filter(a => a.correct).length / attempts.length;

    // IRT predicted probability (2PL model)
    const irtProb = 1 / (1 + Math.exp(-(avgTheta - difficulty)));

    // Consistency score
    const consistency = 1 - Math.abs(correctRate - irtProb);

    // Sample size adjustment
    const sampleAdjustment = Math.min(1, attempts.length / this.config.minAttempts);

    // Quality estimate
    const estimatedQuality = consistency * sampleAdjustment;

    // Confidence based on sample size
    const confidence = Math.min(1, attempts.length / (this.config.minAttempts * 2));

    return {
      questionId,
      estimatedQuality: Math.max(0, Math.min(1, estimatedQuality)),
      confidence,
      isNoisy: estimatedQuality < this.config.noiseThreshold,
    };
  }

  /**
   * Correct label for training based on question quality
   */
  correctLabel(questionId: string, originalLabel: boolean): CorrectedLabel {
    const attempts = this.questionAttempts.get(questionId) ?? [];
    const quality = this.estimateQuality(questionId, attempts, 0); // difficulty not needed for correction

    if (quality.isNoisy) {
      // Use majority vote from high-theta students
      const highThetaAttempts = attempts.filter(a => a.theta > 2.0);
      if (highThetaAttempts.length > 0) {
        const majorityCorrect = highThetaAttempts.filter(a => a.correct).length > highThetaAttempts.length / 2;
        return {
          value: majorityCorrect,
          wasCorrected: true,
          quality: quality.estimatedQuality,
        };
      }
      // Fallback: use model's IRT prediction
      const avgTheta = attempts.reduce((sum, a) => sum + a.theta, 0) / attempts.length;
      const predicted = avgTheta > 0.5;
      return {
        value: predicted,
        wasCorrected: true,
        quality: quality.estimatedQuality,
      };
    }

    return {
      value: originalLabel,
      wasCorrected: false,
      quality: quality.estimatedQuality,
    };
  }

  /**
   * Update model with new response
   */
  update(questionId: string, response: StudentResponse): void {
    const attempts = this.questionAttempts.get(questionId) ?? [];

    // Apply decay to old attempts
    const decayedAttempts = attempts.map(a => ({
      ...a,
      // Decay influence by moving toward 0.5
      correct: Math.random() < this.config.decayRate ? a.correct : Math.random() > 0.5,
    }));

    decayedAttempts.push({ correct: response.correct, theta: response.theta });

    // Limit history size
    if (decayedAttempts.length > this.config.minAttempts * 2) {
      decayedAttempts.shift();
    }

    this.questionAttempts.set(questionId, decayedAttempts);
  }

  /**
   * Get quality for a question
   */
  getQuality(questionId: string): QuestionQuality | undefined {
    const attempts = this.questionAttempts.get(questionId);
    if (!attempts) return undefined;
    return this.estimateQuality(questionId, attempts, 0);
  }
}
```

```typescript
// lib/rl/normalize/feature-normalizer.ts

import type { NormalizerConfig } from '../config/phase3-features';

export interface NormalizationStats {
  mean: number;
  std: number;
  count: number;
}

/**
 * Rolling z-score feature normalizer
 *
 * Uses a rolling window to track statistics and normalize values:
 * normalized = (value - mean) / std
 */
export class FeatureNormalizer {
  private readonly config: NormalizerConfig;
  private windows: Map<string, number[]> = new Map();

  constructor(config: NormalizerConfig) {
    this.config = config;
  }

  /**
   * Normalize a value using rolling statistics
   */
  normalize(value: number, feature: string): number {
    const stats = this.getStats(feature);
    if (stats.count < 2) return 0;
    if (stats.std === 0) return 0;
    return (value - stats.mean) / stats.std;
  }

  /**
   * Denormalize a normalized value back to original scale
   */
  denormalize(normalized: number, feature: string): number {
    const stats = this.getStats(feature);
    if (stats.count < 2) return normalized;
    return normalized * stats.std + stats.mean;
  }

  /**
   * Update with a new value
   */
  update(value: number, feature: string): void {
    const window = this.windows.get(feature) ?? [];
    window.push(value);

    // Enforce window size
    while (window.length > this.config.windowSize) {
      window.shift();
    }

    this.windows.set(feature, window);
  }

  /**
   * Update with multiple values
   */
  updateBatch(values: number[], feature: string): void {
    for (const value of values) {
      this.update(value, feature);
    }
  }

  /**
   * Get current statistics for a feature
   */
  getStats(feature: string): NormalizationStats {
    const window = this.windows.get(feature) ?? [];
    if (window.length === 0) {
      return { mean: 0, std: 0, count: 0 };
    }

    const mean = window.reduce((sum, v) => sum + v, 0) / window.length;
    const variance = window.reduce((sum, v) => sum + Math.pow(v - mean, 2), 0) / window.length;
    const std = Math.sqrt(variance);

    return { mean, std, count: window.length };
  }
}
```

```typescript
// lib/rl/control/adaptation-controller.ts

import type { AdaptationConfig } from '../config/phase3-features';

export interface AdaptationState {
  currentExplorationRate: number;
  confidenceLevel: number;
  learningProgress: number;
  recommendedAction: 'explore' | 'exploit' | 'maintain';
}

export interface HealthMetrics {
  le: number;
  cs: number;
  confidence: number;
}

/**
 * Adaptation Controller - Dynamically adjusts exploration rate
 *
 * Based on:
 * - Confidence: Higher confidence → lower exploration
 * - Learning progress: More learned → lower exploration
 */
export class AdaptationController {
  private readonly config: AdaptationConfig;
  private metrics: HealthMetrics;

  constructor(config: AdaptationConfig) {
    this.config = config;
    this.metrics = { le: 0.15, cs: 0.85, confidence: 0.5 };
  }

  /**
   * Calculate exploration rate based on current state
   */
  calculateExplorationRate(confidence: number, progress: number): number {
    const { baseExplorationRate, minExplorationRate, adaptationSpeed } = this.config;

    // Confidence factor: high confidence → low exploration
    const confidenceFactor = Math.pow(1 - confidence, adaptationSpeed);

    // Progress factor: high progress → low exploration
    const progressFactor = Math.pow(1 - progress, adaptationSpeed * 2);

    const rate = baseExplorationRate * confidenceFactor * progressFactor;

    return Math.max(minExplorationRate, rate);
  }

  /**
   * Get current recommendation
   */
  getRecommendation(): AdaptationState {
    const { confidence, learningProgress } = this.metrics;
    const rate = this.calculateExplorationRate(confidence, learningProgress);

    let recommendedAction: 'explore' | 'exploit' | 'maintain';
    if (confidence < this.config.confidenceThreshold * 0.5) {
      recommendedAction = 'explore';
    } else if (confidence > this.config.confidenceThreshold) {
      recommendedAction = 'exploit';
    } else {
      recommendedAction = 'maintain';
    }

    return {
      currentExplorationRate: rate,
      confidenceLevel: confidence,
      learningProgress,
      recommendedAction,
    };
  }

  /**
   * Update controller with new metrics
   */
  update(metrics: HealthMetrics): void {
    // Weighted update for smoothing
    const smoothing = 0.2;
    this.metrics = {
      le: this.metrics.le * (1 - smoothing) + metrics.le * smoothing,
      cs: this.metrics.cs * (1 - smoothing) + metrics.cs * smoothing,
      confidence: this.metrics.confidence * (1 - smoothing) + metrics.confidence * smoothing,
    };

    // Calculate learning progress from LE
    // LE > 0.15 means learning is happening
    this.metrics.learningProgress = Math.min(1, this.metrics.le / 0.2);
  }
}
```
