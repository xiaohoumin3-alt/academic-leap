🚀 二、自动评分脚本（核心）

import pandas as pd

df = pd.read_csv("test_data.csv")

# -------------------------
# A模块：自适应评分
# -------------------------

def score_adaptive(df):
    total = 0
    correct_rule = 0
    total_cases = 0

    for i in range(len(df) - 1):
        cur = df.iloc[i]
        nxt = df.iloc[i+1]

        # 连续2题正确 → 应升级
        if i >= 1:
            prev = df.iloc[i-1]
            if prev["is_correct"] == 1 and cur["is_correct"] == 1:
                total_cases += 1
                if nxt["difficulty_after"] > cur["difficulty_before"]:
                    correct_rule += 1

        # 错题（非粗心）→ 不升级
        if cur["is_correct"] == 0 and cur["is_careless"] == 0:
            total_cases += 1
            if nxt["difficulty_after"] <= cur["difficulty_before"]:
                correct_rule += 1

    if total_cases == 0:
        return 0

    rate = correct_rule / total_cases

    if rate >= 0.9:
        return 10
    elif rate >= 0.8:
        return 8
    elif rate >= 0.6:
        return 5
    else:
        return 0


# -------------------------
# B模块：估分稳定性（需要两次测评数据）
# -------------------------

def score_estimation(stable_diff):
    # stable_diff = 两次测评分差
    if stable_diff <= 3:
        return 10
    elif stable_diff <= 5:
        return 8
    elif stable_diff <= 8:
        return 5
    else:
        return 0


# -------------------------
# C模块：心流（用时波动 + 正确率波动）
# -------------------------

def score_flow(df):
    correct_rate = df["is_correct"].mean()

    if 0.5 <= correct_rate <= 0.8:
        return 10
    elif 0.4 <= correct_rate <= 0.9:
        return 8
    elif 0.3 <= correct_rate <= 0.95:
        return 5
    else:
        return 0


# -------------------------
# D模块：工程稳定性（简单版）
# -------------------------

def score_stability(df):
    # 这里简单用是否存在异常值代替
    if df["time_used"].isnull().sum() == 0:
        return 20
    else:
        return 10


# -------------------------
# 主评分函数
# -------------------------

def calculate_score(df, estimate_diff):
    A = score_adaptive(df) * 3      # ×3 → 30分
    B = score_estimation(estimate_diff) * 2.5  # ×2.5 → 25分
    C = score_flow(df) * 2.5        # ×2.5 → 25分
    D = score_stability(df)         # 20分

    total = A + B + C + D

    if total >= 85:
        decision = "✅ 可以上线"
    elif total >= 70:
        decision = "⚠️ 灰度上线"
    else:
        decision = "❌ 不建议上线"

    return {
        "A_自适应": A,
        "B_估分": B,
        "C_体验": C,
        "D_稳定性": D,
        "总分": total,
        "结论": decision
    }


# -------------------------
# 使用示例
# -------------------------

# 假设两次测评分差为 4 分
result = calculate_score(df, estimate_diff=4)

print("\n===== 上线评分结果 =====")
for k, v in result.items():
    print(f"{k}: {v}")

import pandas as pd

df = pd.read_csv("test_data.csv")

def score_adaptive(df):
    correct = 0
    total = 0

    for i in range(1, len(df)-1):
        prev = df.iloc[i-1]
        cur = df.iloc[i]
        nxt = df.iloc[i+1]

        if prev["is_correct"] == 1 and cur["is_correct"] == 1:
            total += 1
            if nxt["difficulty_after"] > cur["difficulty_before"]:
                correct += 1

        if cur["is_correct"] == 0 and cur["is_careless"] == 0:
            total += 1
            if nxt["difficulty_after"] <= cur["difficulty_before"]:
                correct += 1

    rate = correct / total if total > 0 else 0

    if rate >= 0.9:
        return 30
    elif rate >= 0.8:
        return 24
    elif rate >= 0.6:
        return 15
    else:
        return 0


def calculate_total(df, estimate_diff):
    A = score_adaptive(df)

    if estimate_diff <= 3:
        B = 25
    elif estimate_diff <= 5:
        B = 20
    else:
        B = 10

    correct_rate = df["is_correct"].mean()
    if 0.5 <= correct_rate <= 0.8:
        C = 25
    else:
        C = 15

    D = 20

    total = A + B + C + D

    return total
